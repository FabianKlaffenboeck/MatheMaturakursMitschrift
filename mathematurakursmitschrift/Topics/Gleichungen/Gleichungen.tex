\section{Gleichungen}

\hfill \break
Wichtige Symbole:
\begin{itemize}
    \item $1=3\ f.A. = falsche aussage (f.A.)$
    \item $3=3\ w.A. = Ware aussage (w.A.)$
    \item 1. Binomische Formel = $(x+y)^2 = x^2+2xy+y^2$
    \item 2. Binomische Formel = $(x-y)^2 = x^2-2xy+y^2$
    \item 3. Binomische Formel = $(x+y)*(x-y) = x^2-y^2$
\end{itemize}

\hfill \break
Vorsilben:
\begin{itemize}
    \item $10^{-12} = piko = p $
    \item $10^{-9} = nano = n $
    \item $10^{-6} = mikro = \mu $
    \item $10^{-3} = mill = m$
    \item $10^{-2} = centi = c$
    \item $10^{-1} = dezi = d$
    \item $10^1 = Deka = da$
    \item $10^2 = Hekto = h$
    \item $10^3 = Kilo = k$
    \item $10^6 = Mega = M = 1 Milion$
    \item $10^9 = Giga = G = 1 Miliarde$
    \item $10^12 = Tera = T = 1 Bilion$
\end{itemize}
