\subsection{logarithmen}

Rechenregeln:
\begin{itemize}
    \item $Log(b^x) = x*Log(b)$
    \item $Log(a*b) = Log(a)+Log(b)$
    \item $Log(\frac{a}{b}) = Log(a)-Log(b)$
    \item $Log(x) = \frac{Log_{10}(x)}{Log_{10}(x)}$
    \item $Log_5(5) = 1$
    \item $Log_{10}(10) = 1$
    \item $Log_e(e) = 1$
    \item $Log_a(a) = 1$
    \item $Log_e(e) = Ln(e)= 1$
