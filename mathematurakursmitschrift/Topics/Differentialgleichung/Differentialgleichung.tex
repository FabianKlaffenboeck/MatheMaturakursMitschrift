\section{Differentialgleichung}

Wichtige ableitungen:
\begin{enumerate}
    \item Potenz-funktionen \begin{itemize}
              \item $f(x) = x^n$
              \item $f'(x) = n*x^{n-1}$
          \end{itemize}
    \item Exponentialfunktionen \begin{itemize}
              \item $f(x) = e^x$
              \item $f'(x) = e^x$
              \item $f(x) = a^x$
              \item $f'(x) = a^x * ln(a)$
          \end{itemize}
    \item Logarithmusfunktion \begin{itemize}
              \item $f(x) = ln(x)$
              \item $f'(x) = \frac{1}{x}$
          \end{itemize}
    \item Trigonometrische Funktionen \begin{itemize}
              \item $f(x) = sin(x)$
              \item $f'(x) = cos(x)$
              \item $f(x) = cos(x)$
              \item $f'(x) = -sin(x)$
              \item $f(x) = tan(x)$
              \item $f'(x) = \frac{1}{cos^2(x)}$
          \end{itemize}
\end{enumerate}
